 StrRScan (function)     (Strings unit)
 ---------------------------------------
Returns a pointer to the last occurrence of a character in a string.

 Declaration:
 function StrRScan(Str: PChar; Chr: Char): PChar;

 Target:
Windows, Real, Protected

 Return Value:
A pointer to the last occurance of Chr in Str
nil if Chr does not occur in Str

 Remarks:
StrRScan returns a pointer to the last occurrence of Chr in Str. If Chr
does not occur in Str, StrRScan returns nil. The null terminator is
considered to be part of the string.

 See Also:
 StrScan

 Sample Code:



 {StrRScan.PAS}

 {Sample code for the StrRScan function.}

 { Return pointer to name part of a full path name }

 { For Windows: }
 { uses Strings, WinCrt; }

 uses Strings;

 function NamePart(FileName: PChar): PChar;
 var
  P: PChar;
 begin
  P := StrRScan(FileName, '\');
  if P = nil then
  begin
    P := StrRScan(FileName, ':');
    if P = nil then P := FileName;
  end;
  NamePart := P;
 end;
 begin
   Writeln (NamePart('C:\Test.fil'));
 end.
