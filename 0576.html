 GetEnvVar (function)     (WinDos unit)
 ---------------------------------------
Returns a pointer to the value of a specified environment variable.

 Declaration:
 function GetEnvVar(var Name: PChar): PChar;

 Target:
Windows

 Return Value:
0 If specified environment variable does not exist

 Sample Code:



 {GetEnvVa.PAS}

 {Sample code for the GetEnvVar function.}

 { For Windows: }
 { uses WinDos, WinCrt; }

 uses WinDos;

 var
  EnvironmentVar: PChar;

 begin
  GetMem(EnvironmentVar, 255);
  EnvironmentVar := GetEnvVar('PATH');
  Writeln('The PATH is currently: ');
  Writeln(EnvironmentVar);
 end.
