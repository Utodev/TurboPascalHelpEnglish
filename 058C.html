 GetViewSettings (procedure)     (Graph unit)
 ---------------------------------------------
Gets the current viewport and clipping parameters.

 Declaration:
 procedure GetViewSettings(var ViewPort: ViewPortType);

 Target:
Real, Protected

 Remarks:
GetViewSettings returns a variable of ViewPortType.

 Restrictions:
Must be in graphics mode.

 See Also:
 ClearViewPort
 SetViewPort

 Sample Code:



 {Getvwset.PAS}

 {Sample code for the GetViewSettings procedure.}

 uses Graph;

 var
  Gd, Gm: Integer;
  ViewPort: ViewPortType;
 begin
  Gd := Detect;
  InitGraph(Gd, Gm, ' ');
  if GraphResult <> grOk then
    Halt(1);
  GetViewSettings(ViewPort);
  with ViewPort do
  begin
    Rectangle(0, 0, X2 - X1, Y2 - Y1);
    if Clip then
      OutText('Clipping is active.')
    else
      OutText('No clipping today.');
  end;
  Readln;
  CloseGraph;
 end.
