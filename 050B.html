 StrScan (function)     (Strings unit)
 --------------------------------------
Returns a pointer to the first occurrence of a character in a string.

 Declaration:
 function StrScan(Str: PChar; Chr: Char): PChar;

 Target:
Windows, Real, Protected

 Return Value:
0 if Chr does not occur in Str

 Remarks:
StrScan returns a pointer to the first occurrence of Chr in Str. If Chr
does not occur in Str, StrScan returns nil. The null terminator is
considered to be part of the string.

 See Also:
 StrRScan

 Sample Code:



 {StrScan.PAS}

 {Sample code for the StrScan function.}

 { Return true if file name has wildcards in it }

 { For Windows: }
 { uses WinCrt, Strings; }

 uses Strings;

 function HasWildcards(FileName: PChar): Boolean;
 begin
  HasWildcards := (StrScan(FileName, '*') <> nil) or
    (StrScan(FileName, '?') <> nil);
 end;
 begin
   Writeln ('Has wildcards: ', HasWildCards('C:\Test.*'));
 end.
